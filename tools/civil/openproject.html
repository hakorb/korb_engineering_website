docker-compose.yml 
# ===========================================================================
# OpenProject Community Edition — Production Docker Compose
# ===========================================================================

# Shared configuration anchors
x-openproject-common: &openproject-common
  image: openproject/openproject:15-slim
  restart: unless-stopped
  env_file:
    - .env
  environment:
    DATABASE_URL: "postgres://${POSTGRES_USER}:${POSTGRES_PASSWORD}@postgres:5432/${POSTGRES_DB}?pool=20"
    OPENPROJECT_HTTPS: "${OPENPROJECT_HTTPS:-false}"
    OPENPROJECT_HOST__NAME: "${OPENPROJECT_HOST__NAME}"
    SECRET_KEY_BASE: "${SECRET_KEY_BASE}"
    RAILS_CACHE_STORE: "file_store"
    # ... plus SMTP, logging, locale vars
  volumes:
    - opdata:/var/openproject/assets
  networks:
    - openproject-net
  logging:
    driver: json-file
    options:
      max-size: "10m"
      max-file: "3"

services:

  # PostgreSQL 17
    postgres:
    image: postgres:17-alpine
    container_name: openproject-postgres
    restart: unless-stopped
    volumes:
      - pgdata:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_USER}"]
      interval: 10s
      timeout: 5s
      retries: 5
      start_period: 30s

  # Puma Web Server
    web:
    <<: *openproject-common
    container_name: openproject-web
    command: "./docker/prod/web"
    ports:
      - "127.0.0.1:${PORT:-8080}:8080"
    depends_on:
      postgres: { condition: service_healthy }
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/health_checks/default"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 90s

  # Background Worker
    worker:
    <<: *openproject-common
    command: "./docker/prod/worker"
    depends_on:
      postgres: { condition: service_healthy }
      web: { condition: service_started }

  # Cron Scheduler
    cron:
    <<: *openproject-common
    command: "./docker/prod/cron"
    depends_on:
      postgres: { condition: service_healthy }
      web: { condition: service_started }

volumes:
  pgdata:  # bind → ./pgdata
  opdata:  # bind → ./assets

networks:
  openproject-net:
    driver: bridge

OPENPROJECT_SMTP__ENABLE__STARTTLS__AUTO=true
OPENPROJECT_MAIL__FROM=openproject@example.com

🚀 Deployment Steps

9 steps
1
Copy environment template
Create your local environment file from the example template.
cp .env.example .env
2
Generate secrets
Create cryptographically secure keys for Rails and PostgreSQL.
openssl rand -hex 64   # SECRET_KEY_BASE

openssl rand -hex 32   # POSTGRES_PASSWORD
3
Set your domain
Update OPENPROJECT_HOST__NAME in .env to your public domain, or use localhost:8080 for local development.
4
Configure SMTP
Fill in SMTP settings for email notifications (optional for dev, required for production).
5
Launch all services
Start PostgreSQL, Web, Worker, and Cron containers in detached mode.
docker compose up -d
6
Verify containers
Check that all 4 containers are running and healthy.
docker compose ps
7
Watch startup logs
Follow the web server log output to confirm Rails boot completes without errors.
docker compose logs -f web
8
Open in browser
Navigate to your configured host to access the OpenProject login screen.
http://localhost:8080
9
Change admin password
Login with admin / admin and immediately change the default password.
⚠ The default admin password must be changed on first login for security.

💻 Common Commands

Lifecycle 
# Start all services
docker compose up -d

# Stop all services
docker compose down

# Restart a single service
docker compose restart web

# View status
docker compose ps

# Follow logs
docker compose logs -f web
docker compose logs -f worker
Updates 
# Pull latest images
docker compose pull

# Recreate with new images
docker compose up -d --force-recreate

# Clean up old images
docker image prune -f

🔧 Maintenance

Database 
# Backup database
docker compose exec postgres \
  pg_dump -U openproject openproject \
  > backup_$(date +%Y%m%d).sql

# Restore database
docker compose exec -T postgres \
  psql -U openproject openproject \
  < backup_20260328.sql

# Access psql console
docker compose exec postgres \
  psql -U openproject openproject
Rails Console 
# Open Rails console
docker compose exec web \
  bundle exec rails console

# Run migrations manually
docker compose exec web \
  bundle exec rails db:migrate

# Re-seed (first run only)
docker compose exec web \
  bundle exec rails db:seed
